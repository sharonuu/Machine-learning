class LR_GD():
    def _init_(self):
        self.w = None
    def fit(self,X,y,alpha = 0.02,loss = 1e-10):
        y = y.reshape(-1,1)
        [m,d] = np.shape(X)
        self.w = np.zeros((d))
        tol = 1e5
        if tol >loss:
            told= y-np.dot(X,self.w)
            self.w = self.w - 1/m *alpha * (np.dot(told , X))
            tol = np.abs(np.sum(told))
    def predict(self,X):
        y_pred = np.dot(X,self.w)
        return y_pred
if __name__ == "__main__":

class LR_LS():
    def _init_(self):
        self.w = None
    def fit(self,X,y):
        temp0 = np.dot(X.T,X)
        temp = np.dot(np.linalg.inv(temp0),X.T)
        self.w = np.dot(temp,y)
    def predict(self,X):
        y_pred = np.dot(X,self.w)
        return y_pred
if __name__ == "__main__":
